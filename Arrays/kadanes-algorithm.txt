class Solution {
    public int maxSubArray(int[] nums) {
        
        int sum = nums[nums.length - 1];
        int msumi = nums[nums.length - 1];
        
        
        
        for(int i = nums.length - 2 ; i >= 0 ; i--) {
            int pmax = Math.max(nums[i], nums[i] + msumi);
            msumi = pmax;
            if(msumi > sum) {
                sum = msumi;
            }
            
        }
        
        return sum;
    }
}
