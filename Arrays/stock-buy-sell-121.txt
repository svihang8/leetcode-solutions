class Solution {
    public int maxProfit(int[] prices) {
        int profit = 0;
        int buy = prices[prices.length - 1];
        int sell = prices[prices.length - 1];
        
        for(int i = prices.length - 1 ; i >= 0 ; i--) {
            buy = prices[i];
            sell = Math.max(sell, prices[i]);
            profit = Math.max(profit, sell - buy);
        }
        
        return profit;
        
        
    }
}
